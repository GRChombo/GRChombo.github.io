46 template <class potential_t>
47 template <class data_t, template <typename> class vars_t>
48 void ScalarField<potential_t>::emtensor_excl_potential(
49  emtensor_t<data_t> &out, const vars_t<data_t> &vars,
50  const SFObject<data_t> &vars_sf, const Tensor<1, data_t> &d1_phi,
51  const Tensor<2, data_t> &h_UU, const Tensor<3, data_t> &chris_ULL)
52 {
53  // Useful quantity Vt
54  data_t Vt = -vars_sf.Pi * vars_sf.Pi;
55  FOR2(i, j) { Vt += vars.chi * h_UU[i][j] * d1_phi[i] * d1_phi[j]; }
56 
57  // Calculate components of EM Tensor
58  // S_ij = T_ij
59  FOR2(i, j)
60  {
61  out.Sij[i][j] =
62  -0.5 * vars.h[i][j] * Vt / vars.chi + d1_phi[i] * d1_phi[j];
63  }
64 
65  // S = Tr_S_ij
66  out.S = vars.chi * TensorAlgebra::compute_trace(out.Sij, h_UU);
67 
68  // S_i (note lower index) = n^a T_a0
69  FOR1(i) { out.Si[i] = -d1_phi[i] * vars_sf.Pi; }
70 
71  // rho = n^a n^b T_ab
72  out.rho = vars_sf.Pi * vars_sf.Pi + 0.5 * Vt;
73 }

116 // the RHS excluding the potential terms
117 template <class potential_t>
118 template <class data_t, template <typename> class vars_t>
119 void ScalarField<potential_t>::matter_rhs_excl_potential(
120  SFObject<data_t> &rhs_sf, const vars_t<data_t> &vars,
121  const SFObject<data_t> &vars_sf, const vars_t<Tensor<1, data_t>> &d1,
122  const Tensor<1, data_t> &d1_phi, const Tensor<2, data_t> &d2_phi,
123  const SFObject<data_t> &advec_sf)
124 {
125  using namespace TensorAlgebra;
126 
127  const auto h_UU = compute_inverse_sym(vars.h);
128  const auto chris = compute_christoffel(d1.h, h_UU);
129 
130  // evolution equations for scalar field and (minus) its conjugate momentum
131  rhs_sf.phi = vars.lapse * vars_sf.Pi + advec_sf.phi;
132  rhs_sf.Pi = vars.lapse * vars.K * vars_sf.Pi + advec_sf.Pi;
133 
134  FOR2(i, j)
135  {
136  // includes non conformal parts of chris not included in chris_ULL
137  rhs_sf.Pi += h_UU[i][j] * (-0.5 * d1.chi[j] * vars.lapse * d1_phi[i] +
138  vars.chi * vars.lapse * d2_phi[i][j] +
139  vars.chi * d1.lapse[i] * d1_phi[j]);
140  FOR1(k)
141  {
142  rhs_sf.Pi += -vars.chi * vars.lapse * h_UU[i][j] *
143  chris.ULL[k][i][j] * d1_phi[k];
144  }
145  }
146 }
