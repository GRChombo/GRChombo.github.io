36 void mainSetup(int argc, char *argv[])
37 {
38 #ifdef CH_MPI
39  // Start MPI
40  MPI_Init(&argc, &argv);
41 #ifdef CH_AIX
42  H5dont_atexit();
43 #endif
44 // setChomboMPIErrorHandler();
45 #endif
46 
47  int rank, number_procs;
48 #ifdef CH_MPI
49  MPI_Comm_rank(Chombo_MPI::comm, &rank);
50  MPI_Comm_size(Chombo_MPI::comm, &number_procs);
51 #else
52  rank = 0;
53  number_procs = 1;
54 #endif
55 
56 #ifdef EQUATION_DEBUG_MODE
57  EquationDebugging::check_no_omp();
58  MayDay::Warning("GRChombo is running in equation debug mode. This mode is "
59  "intended only for debugging and leads to significantly "
60  "worse performance.");
61 #endif
62 
63  if (rank == 0)
64  {
65  pout() << " number_procs = " << number_procs << endl;
66 #ifdef _OPENMP
67  pout() << " threads = " << omp_get_max_threads() << endl;
68 #endif
69  }
70 
71  const int required_argc = 2;
72  if (argc < required_argc)

88 void setupAMRObject(GRAMR &gr_amr, AMRLevelFactory &a_factory)
89 {
90  // Some hard-coded parameters:
91  // The buffer is width of ghost cells + additional_grid_buffer
92  // and defines the minimum number of level l cells there have to be
93  // between level l+1 and level l-1
94  const int additional_grid_buffer = 3;
95 
96  ParmParse pp;
97 
98  IntVect ivN = IntVect::Unit;
99  // Setup the grid size
100  for (int dir = 0; dir < SpaceDim; ++dir)
101  {
102  char dir_str[20];
103  sprintf(dir_str, "N%d", dir + 1);
104  int N;
105  pp.get(dir_str, N);
106  ivN[dir] = N - 1;
107  }
108 
109  Box problem_domain(IntVect::Zero, ivN);
110  ProblemDomain physdomain(problem_domain);
111 
112  // set periodicity
113  std::vector<bool> isPeriodic;
114  pp.getarr("isPeriodic", isPeriodic, 0, SpaceDim);
115  for (int dir = 0; dir < SpaceDim; dir++)
116  {
117  physdomain.setPeriodic(dir, isPeriodic[dir]);
118  }
119 
120  int max_level;
121  pp.get("max_level", max_level);
122 
123  Vector<int> ref_ratios;
124  pp.getarr("ref_ratio", ref_ratios, 0, max_level + 1);
125 
126  // Define the AMR object
127  gr_amr.define(max_level, ref_ratios, physdomain, &a_factory);
128 
129  // To preserve proper nesting we need to know the maximum ref_ratio.
130  int max_ref_ratio = ref_ratios[0];
131  for (int i = 1; i < max_level + 1; ++i)
132  {
133  max_ref_ratio = std::max(max_ref_ratio, ref_ratios[i]);
134  }
135 
136  int num_ghosts;
137  pp.get("num_ghosts", num_ghosts);
138  int grid_buffer_size =
139  std::ceil(((double)num_ghosts) / (double)max_ref_ratio) +
140  additional_grid_buffer;
141  gr_amr.gridBufferSize(grid_buffer_size);
142 
143  int checkpoint_interval;
144  pp.get("checkpoint_interval", checkpoint_interval);
145  gr_amr.checkpointInterval(checkpoint_interval);
146 
147  // Number of coarse time steps from one regridding to the next
148  Vector<int> regrid_intervals;
149  pp.getarr("regrid_interval", regrid_intervals, 0, max_level + 1);
150  gr_amr.regridIntervals(regrid_intervals);
151 
152  if (pp.contains("max_grid_size"))
153  {
154  int max_grid_size;
155  pp.query("max_grid_size", max_grid_size);
156  gr_amr.maxGridSize(max_grid_size);
157  }
158 
159  if (pp.contains("block_factor"))
160  {
161  int block_factor;
162  pp.query("block_factor", block_factor);
163  gr_amr.blockFactor(block_factor);
164  }
165 
166  if (pp.contains("fill_ratio"))
167  {
168  Real fill_ratio;
169  pp.query("fill_ratio", fill_ratio);
170  gr_amr.fillRatio(fill_ratio);
171  }
172 
173  if (pp.contains("max_dt_grow"))
174  {
175  Real max_dt_grow;
176  pp.query("max_dt_grow", max_dt_grow);
177  gr_amr.maxDtGrow(max_dt_grow);
178  }
179 
180  if (pp.contains("dt_tolerance_factor"))
181  {
182  Real dt_tolerance_factor;
183  pp.query("dt_tolerance_factor", dt_tolerance_factor);
184  gr_amr.dtToleranceFactor(dt_tolerance_factor);
185  }
186 
187  if (pp.contains("chk_prefix"))
188  {
189  std::string prefix;
190  pp.query("chk_prefix", prefix);
191  gr_amr.checkpointPrefix(prefix);
192  }
193 
194  if (pp.contains("plot_interval"))
195  {
196  int plot_interval;
197  pp.get("plot_interval", plot_interval);
198  gr_amr.plotInterval(plot_interval);
199  }
200 
201  if (pp.contains("plot_prefix"))
202  {
203  std::string prefix;
204  pp.query("plot_prefix", prefix);
205  gr_amr.plotPrefix(prefix);
206  }
207 
208  int verbosity;
209  pp.get("verbosity", verbosity);
210  gr_amr.verbosity(verbosity);
211 
212  // Set up input files
213  if (!pp.contains("restart_file"))
214  {
215  gr_amr.setupForNewAMRRun();
216  }
217  else
218  {
219  std::string restart_file;
220  pp.query("restart_file", restart_file);
221 
222 #ifdef CH_USE_HDF5
223  HDF5Handle handle(restart_file, HDF5Handle::OPEN_RDONLY);
224  // read from checkpoint file
225  gr_amr.setupForRestart(handle);
226  handle.close();
227 #else
228  MayDay::Error("GRChombo restart only defined with hdf5");
229 #endif
230  }
231 }
