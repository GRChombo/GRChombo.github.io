23 template <class data_t>
24 ALWAYS_INLINE data_t compute_determinant_sym(const Tensor<2, data_t, 3> &matrix)
25 {
26  data_t det = matrix[0][0] * matrix[1][1] * matrix[2][2] +
27  2 * matrix[0][1] * matrix[0][2] * matrix[1][2] -
28  matrix[0][0] * matrix[1][2] * matrix[1][2] -
29  matrix[1][1] * matrix[0][2] * matrix[0][2] -
30  matrix[2][2] * matrix[0][1] * matrix[0][1];
31 
32  return det;
33 }

46  (matrix[1][0] * matrix[2][1] - matrix[1][1] * matrix[2][0]);

51 template <class data_t>
52 Tensor<2, data_t> compute_inverse_sym(const Tensor<2, data_t, 3> &matrix)
53 {
54  data_t deth = compute_determinant_sym(matrix);
55  data_t deth_inverse = 1. / deth;
56  Tensor<2, data_t> h_UU;
57  h_UU[0][0] = (matrix[1][1] * matrix[2][2] - matrix[1][2] * matrix[1][2]) *
58  deth_inverse;
59  h_UU[0][1] = (matrix[0][2] * matrix[1][2] - matrix[0][1] * matrix[2][2]) *
60  deth_inverse;
61  h_UU[0][2] = (matrix[0][1] * matrix[1][2] - matrix[0][2] * matrix[1][1]) *
62  deth_inverse;
63  h_UU[1][1] = (matrix[0][0] * matrix[2][2] - matrix[0][2] * matrix[0][2]) *
64  deth_inverse;
65  h_UU[1][2] = (matrix[0][1] * matrix[0][2] - matrix[0][0] * matrix[1][2]) *
66  deth_inverse;
67  h_UU[2][2] = (matrix[0][0] * matrix[1][1] - matrix[0][1] * matrix[0][1]) *
68  deth_inverse;
69  h_UU[1][0] = h_UU[0][1];
70  h_UU[2][0] = h_UU[0][2];
71  h_UU[2][1] = h_UU[1][2];
72 
73  return h_UU;
74 }

237 template <class data_t>
238 chris_t<data_t>
239 compute_christoffel(const Tensor<2, Tensor<1, data_t>> &d1_metric,
240  const Tensor<2, data_t> &h_UU)
241 {
242  chris_t<data_t> out;
243 
244  FOR3(i, j, k)
245  {
246  out.LLL[i][j][k] = 0.5 * (d1_metric[j][i][k] + d1_metric[k][i][j] -
247  d1_metric[j][k][i]);
248  }
249  FOR3(i, j, k)
250  {
251  out.ULL[i][j][k] = 0;
252  FOR1(l) { out.ULL[i][j][k] += h_UU[i][l] * out.LLL[l][j][k]; }
253  }
254  FOR1(i)
255  {
256  out.contracted[i] = 0;
257  FOR2(j, k) { out.contracted[i] += h_UU[j][k] * out.ULL[i][j][k]; }
258  }
259 
260  return out;
261 }
