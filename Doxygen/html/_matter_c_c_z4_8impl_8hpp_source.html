22 template <class matter_t>
23 template <class data_t>
24 void MatterCCZ4<matter_t>::compute(Cell<data_t> current_cell) const
25 {
26  // copy data from chombo gridpoint into local variables
27  const auto matter_vars = current_cell.template load_vars<Vars>();
28  const auto d1 = m_deriv.template diff1<Vars>(current_cell);
29  const auto d2 = m_deriv.template diff2<Diff2Vars>(current_cell);
30  const auto advec =
31  m_deriv.template advection<Vars>(current_cell, matter_vars.shift);
32 
33  // Call CCZ4 RHS - work out RHS without matter, no dissipation
34  Vars<data_t> matter_rhs;
35  rhs_equation(matter_rhs, matter_vars, d1, d2, advec);
36 
37  // add RHS matter terms from EM Tensor
38  add_emtensor_rhs(matter_rhs, matter_vars, d1);
39 
40  // add evolution of matter fields themselves
41  my_matter.add_matter_rhs(matter_rhs, matter_vars, d1, d2, advec);
42 
43  // Add dissipation to all terms
44  m_deriv.add_dissipation(matter_rhs, current_cell, m_sigma);
45 
46  // Write the rhs into the output FArrayBox
47  current_cell.store_vars(matter_rhs);
48 }

50 // Function to add in EM Tensor matter terms to CCZ4 rhs
51 template <class matter_t>
52 template <class data_t>
53 void MatterCCZ4<matter_t>::add_emtensor_rhs(
54  Vars<data_t> &matter_rhs, const Vars<data_t> &matter_vars,
55  const Vars<Tensor<1, data_t>> &d1) const
56 {
57  using namespace TensorAlgebra;
58 
59  const auto h_UU = compute_inverse_sym(matter_vars.h);
60  const auto chris = compute_christoffel(d1.h, h_UU);
61 
62  // Calculate elements of the decomposed stress energy tensor
63  const auto emtensor =
64  my_matter.compute_emtensor(matter_vars, d1, h_UU, chris.ULL);
65 
66  // Update RHS for K and Theta depending on formulation
67  if (m_formulation == USE_BSSN)
68  {
69  matter_rhs.K += 4.0 * M_PI * m_G_Newton * matter_vars.lapse *
70  (emtensor.S + emtensor.rho);
71  matter_rhs.Theta += 0.0;
72  }
73  else
74  {
75  matter_rhs.K += 4.0 * M_PI * m_G_Newton * matter_vars.lapse *
76  (emtensor.S - 3 * emtensor.rho);
77  matter_rhs.Theta +=
78  -8.0 * M_PI * m_G_Newton * matter_vars.lapse * emtensor.rho;
79  }
80 
81  // Update RHS for other variables
82  Tensor<2, data_t> Sij_TF = emtensor.Sij;
83  make_trace_free(Sij_TF, matter_vars.h, h_UU);
84 
85  FOR2(i, j)
86  {
87  matter_rhs.A[i][j] += -8.0 * M_PI * m_G_Newton * matter_vars.chi *
88  matter_vars.lapse * Sij_TF[i][j];
89  }
90 
91  FOR1(i)
92  {
93  data_t matter_term_Gamma = 0.0;
94  FOR1(j)
95  {
96  matter_term_Gamma += -16.0 * M_PI * m_G_Newton * matter_vars.lapse *
97  h_UU[i][j] * emtensor.Si[j];
98  }
99 
100  matter_rhs.Gamma[i] += matter_term_Gamma;
101  matter_rhs.B[i] += matter_term_Gamma;
102  }
103 }
