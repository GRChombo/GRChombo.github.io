18 inline CCZ4::CCZ4(params_t params, double dx, double sigma, int formulation,
19  double cosmological_constant)
20  : m_params(params), m_sigma(sigma), m_formulation(formulation),
21  m_cosmological_constant(cosmological_constant), m_deriv(dx)
22 {
23  // A user who wants to use BSSN should also have damping paramters = 0

27  (params.kappa3 != 0.))
28  {

30  "should be set to zero in params");
31  }
32  }

35 }

37 template <class data_t> void CCZ4::compute(Cell<data_t> current_cell) const
38 {
39  const auto vars = current_cell.template load_vars<Vars>();
40  const auto d1 = m_deriv.template diff1<Vars>(current_cell);
41  const auto d2 = m_deriv.template diff2<Diff2Vars>(current_cell);
42  const auto advec =
43  m_deriv.template advection<Vars>(current_cell, vars.shift);
44 
45  Vars<data_t> rhs;
46  rhs_equation(rhs, vars, d1, d2, advec);
47 
48  m_deriv.add_dissipation(rhs, current_cell, m_sigma);
49 
50  current_cell.store_vars(rhs); // Write the rhs into the output FArrayBox
51 }

53 template <class data_t, template <typename> class vars_t,
54  template <typename> class diff2_vars_t>
55 void CCZ4::rhs_equation(vars_t<data_t> &rhs, const vars_t<data_t> &vars,
56  const vars_t<Tensor<1, data_t>> &d1,
57  const diff2_vars_t<Tensor<2, data_t>> &d2,
58  const vars_t<data_t> &advec) const
59 {
60  using namespace TensorAlgebra;
61 
62  auto h_UU = compute_inverse_sym(vars.h);
63  auto chris = compute_christoffel(d1.h, h_UU);
64 
65  Tensor<1, data_t> Z_over_chi;
66  Tensor<1, data_t> Z;
67 
68  if (m_formulation == USE_BSSN)
69  {
70  FOR1(i) Z_over_chi[i] = 0.0;
71  }
72  else
73  {
74  FOR1(i) Z_over_chi[i] = 0.5 * (vars.Gamma[i] - chris.contracted[i]);
75  }
76  FOR1(i) Z[i] = vars.chi * Z_over_chi[i];
77 
78  auto ricci =
79  CCZ4Geometry::compute_ricci_Z(vars, d1, d2, h_UU, chris, Z_over_chi);
80 
81  data_t divshift = compute_trace(d1.shift);
82  data_t Z_dot_d1lapse = compute_dot_product(Z, d1.lapse);
83  data_t dlapse_dot_dchi = compute_dot_product(d1.lapse, d1.chi, h_UU);
84 
85  Tensor<2, data_t> covdtilde2lapse;
86  Tensor<2, data_t> covd2lapse;
87  FOR2(k, l)
88  {
89  covdtilde2lapse[k][l] = d2.lapse[k][l];
90  FOR1(m) { covdtilde2lapse[k][l] -= chris.ULL[m][k][l] * d1.lapse[m]; }
91  covd2lapse[k][l] =
92  vars.chi * covdtilde2lapse[k][l] +
93  0.5 * (d1.lapse[k] * d1.chi[l] + d1.chi[k] * d1.lapse[l] -
94  vars.h[k][l] * dlapse_dot_dchi);
95  }
96 
97  data_t tr_covd2lapse = -(GR_SPACEDIM / 2.0) * dlapse_dot_dchi;
98  FOR1(i)
99  {
100  tr_covd2lapse -= vars.chi * chris.contracted[i] * d1.lapse[i];
101  FOR1(j)
102  {
103  tr_covd2lapse += h_UU[i][j] * (vars.chi * d2.lapse[i][j] +
104  d1.lapse[i] * d1.chi[j]);
105  }
106  }
107 
108  Tensor<2, data_t> A_UU = raise_all(vars.A, h_UU);
109 
110  // A^{ij} A_{ij}. - Note the abuse of the compute trace function.
111  data_t tr_A2 = compute_trace(vars.A, A_UU);
112  rhs.chi = advec.chi +
113  (2.0 / GR_SPACEDIM) * vars.chi * (vars.lapse * vars.K - divshift);
114  FOR2(i, j)
115  {
116  rhs.h[i][j] = advec.h[i][j] - 2.0 * vars.lapse * vars.A[i][j] -
117  (2.0 / GR_SPACEDIM) * vars.h[i][j] * divshift;
118  FOR1(k)
119  {
120  rhs.h[i][j] +=
121  vars.h[k][i] * d1.shift[k][j] + vars.h[k][j] * d1.shift[k][i];
122  }
123  }
124 
125  Tensor<2, data_t> Adot_TF;
126  FOR2(i, j)
127  {
128  Adot_TF[i][j] =
129  -covd2lapse[i][j] + vars.chi * vars.lapse * ricci.LL[i][j];
130  }
131  make_trace_free(Adot_TF, vars.h, h_UU);
132 
133  FOR2(i, j)
134  {
135  rhs.A[i][j] = advec.A[i][j] + Adot_TF[i][j] +
136  vars.A[i][j] * (vars.lapse * (vars.K - 2 * vars.Theta) -
137  (2.0 / GR_SPACEDIM) * divshift);
138  FOR1(k)
139  {
140  rhs.A[i][j] +=
141  vars.A[k][i] * d1.shift[k][j] + vars.A[k][j] * d1.shift[k][i];
142  FOR1(l)
143  {
144  rhs.A[i][j] -=
145  2 * vars.lapse * h_UU[k][l] * vars.A[i][k] * vars.A[l][j];
146  }
147  }
148  }
149 
150 #ifdef COVARIANTZ4
151  data_t kappa1_lapse = m_params.kappa1;
152 #else
153  data_t kappa1_lapse = m_params.kappa1 * vars.lapse;
154 #endif
155 
156  if (m_formulation == USE_BSSN)
157  {
158  rhs.Theta = 0; // ensure the Theta of CCZ4 remains at zero
159  // Use hamiltonian constraint to remove ricci.scalar for BSSN update
160  rhs.K = advec.K + vars.lapse * (tr_A2 + vars.K * vars.K / GR_SPACEDIM) -
161  tr_covd2lapse;
162  rhs.K += -2 * vars.lapse * m_cosmological_constant / (GR_SPACEDIM - 1.);
163  }
164  else
165  {
166  rhs.Theta =
167  advec.Theta +
168  0.5 * vars.lapse *
169  (ricci.scalar - tr_A2 +
170  ((GR_SPACEDIM - 1.0) / (double)GR_SPACEDIM) * vars.K * vars.K -
171  2 * vars.Theta * vars.K) -
172  0.5 * vars.Theta * kappa1_lapse *
173  ((GR_SPACEDIM + 1) + m_params.kappa2 * (GR_SPACEDIM - 1)) -
174  Z_dot_d1lapse;
175 
176  rhs.Theta += -vars.lapse * m_cosmological_constant;
177  rhs.K =
178  advec.K +
179  vars.lapse * (ricci.scalar + vars.K * (vars.K - 2 * vars.Theta)) -
180  kappa1_lapse * GR_SPACEDIM * (1 + m_params.kappa2) * vars.Theta -
181  tr_covd2lapse;
182  rhs.K += -2 * vars.lapse * GR_SPACEDIM / (GR_SPACEDIM - 1.) *
183  m_cosmological_constant;
184  }
185 
186  Tensor<1, data_t> Gammadot;
187  FOR1(i)
188  {
189  Gammadot[i] = (2.0 / GR_SPACEDIM) *
190  (divshift * (chris.contracted[i] +
191  2 * m_params.kappa3 * Z_over_chi[i]) -
192  2 * vars.lapse * vars.K * Z_over_chi[i]) -
193  2 * kappa1_lapse * Z_over_chi[i];
194  FOR1(j)
195  {
196  Gammadot[i] +=
197  2 * h_UU[i][j] *
198  (vars.lapse * d1.Theta[j] - vars.Theta * d1.lapse[j]) -
199  2 * A_UU[i][j] * d1.lapse[j] -
200  vars.lapse * ((2 * (GR_SPACEDIM - 1.0) / (double)GR_SPACEDIM) *
201  h_UU[i][j] * d1.K[j] +
202  GR_SPACEDIM * A_UU[i][j] * d1.chi[j] / vars.chi) -
203  (chris.contracted[j] + 2 * m_params.kappa3 * Z_over_chi[j]) *
204  d1.shift[i][j];
205 
206  FOR1(k)
207  {
208  Gammadot[i] +=
209  2 * vars.lapse * chris.ULL[i][j][k] * A_UU[j][k] +
210  h_UU[j][k] * d2.shift[i][j][k] +
211  ((GR_SPACEDIM - 2.0) / (double)GR_SPACEDIM) * h_UU[i][j] *
212  d2.shift[k][j][k];
213  }
214  }
215  }
216 
217  FOR1(i) { rhs.Gamma[i] = advec.Gamma[i] + Gammadot[i]; }
218 
219  const data_t etaDecay = 1.;
220 
221  rhs.lapse = m_params.lapse_advec_coeff * advec.lapse -
222  m_params.lapse_coeff * pow(vars.lapse, m_params.lapse_power) *
223  (vars.K - 2 * vars.Theta);
224  FOR1(i)
225  {
226  rhs.shift[i] = m_params.shift_advec_coeff * advec.shift[i] +
227  m_params.shift_Gamma_coeff * vars.B[i];
228  rhs.B[i] = m_params.shift_advec_coeff * advec.B[i] +
229  (1 - m_params.shift_advec_coeff) * advec.Gamma[i] +
230  Gammadot[i] - m_params.eta * etaDecay * vars.B[i];
231  }
232 }

247  define_enum_mapping(mapping_function, c_lapse, lapse);
248 
249  // Vectors

251  Gamma);

253  shift);
254  define_enum_mapping(mapping_function, GRInterval<c_B1, c_B3>(), B);
255 
256  // Symmetric 2-tensors

258  h);
259  define_symmetric_enum_mapping(mapping_function, GRInterval<c_A11, c_A33>(),
260  A);
