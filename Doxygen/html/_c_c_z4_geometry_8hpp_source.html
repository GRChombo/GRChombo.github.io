31  public:
32  template <class data_t, template <typename> class vars_t,
33  template <typename> class diff2_vars_t>
34  static ricci_t<data_t>
35  compute_ricci_Z(const vars_t<data_t> &vars,
36  const vars_t<Tensor<1, data_t>> &d1,
37  const diff2_vars_t<Tensor<2, data_t>> &d2,
38  const Tensor<2, data_t> &h_UU, const chris_t<data_t> &chris,
39  const Tensor<1, data_t> &Z_over_chi)
40  {
41  ricci_t<data_t> out;
42 
43  data_t boxtildechi = 0;
44 
45  Tensor<2, data_t> covdtilde2chi;
46  FOR2(k, l)
47  {
48  covdtilde2chi[k][l] = d2.chi[k][l];
49  FOR1(m) { covdtilde2chi[k][l] -= chris.ULL[m][k][l] * d1.chi[m]; }
50  }
51 
52  FOR2(k, l) { boxtildechi += h_UU[k][l] * covdtilde2chi[k][l]; }
53 
54  data_t dchi_dot_dchi = 0;
55  {
56  FOR2(m, n) { dchi_dot_dchi += h_UU[m][n] * d1.chi[m] * d1.chi[n]; }
57  }
58 
59  FOR2(i, j)
60  {
61  data_t ricci_tilde = 0;
62  FOR1(k)
63  {
64  // Trick: For CCZ4, we can add Z terms to ricci by changing
65  // Gamma to chrisvec This way of writing it allows the user to
66  // pass Z/chi = {0};
67  ricci_tilde += 0.5 * (vars.h[k][i] * d1.Gamma[k][j] +
68  vars.h[k][j] * d1.Gamma[k][i]);
69  ricci_tilde += 0.5 * (vars.Gamma[k] - 2 * Z_over_chi[k]) *
70  (chris.LLL[i][j][k] + chris.LLL[j][i][k]);
71  FOR1(l)
72  {
73  ricci_tilde -= 0.5 * h_UU[k][l] * d2.h[i][j][k][l];
74  FOR1(m)
75  {
76  ricci_tilde +=
77  h_UU[l][m] *
78  (chris.ULL[k][l][i] * chris.LLL[j][k][m] +
79  chris.ULL[k][l][j] * chris.LLL[i][k][m] +
80  chris.ULL[k][i][m] * chris.LLL[k][l][j]);
81  }
82  }
83  }
84 
85  data_t ricci_chi =
86  0.5 * ((GR_SPACEDIM - 2) * covdtilde2chi[i][j] +
87  vars.h[i][j] * boxtildechi -
88  ((GR_SPACEDIM - 2) * d1.chi[i] * d1.chi[j] +
89  GR_SPACEDIM * vars.h[i][j] * dchi_dot_dchi) /
90  (2 * vars.chi));
91 
92  data_t z_terms = 0;
93  FOR1(k)
94  {
95  z_terms +=
96  Z_over_chi[k] *
97  (vars.h[i][k] * d1.chi[j] + vars.h[j][k] * d1.chi[i] -
98  vars.h[i][j] * d1.chi[k] + d1.h[i][j][k] * vars.chi);
99  }
100 
101  out.LL[i][j] =
102  (ricci_chi + vars.chi * ricci_tilde + z_terms) / vars.chi;
103  }
104 
105  out.scalar = vars.chi * TensorAlgebra::compute_trace(out.LL, h_UU);
106 
107  return out;
108  }
