61 template <typename InterpAlgo>
62 void AMRInterpolator<InterpAlgo>::interp(InterpolationQuery &query)
63 {
64  if (m_verbosity)
65  {
66  pout() << TAG << "\x1b[32;1mInterpolating data\x1b[0m" << endl;
67 
68  for (typename InterpolationQuery::iterator it = query.compsBegin();
69  it != query.compsEnd(); ++it)
70  {
71  const Derivative deriv = it->first;
72 
73  pout() << " This rank is querying for interpolated D(";
74  for (int i = 0; i < CH_SPACEDIM; ++i)
75  {
76  pout() << deriv[i];
77  if (i < CH_SPACEDIM - 1)
78  {
79  pout() << ",";
80  }
81  }
82  pout() << ") data for " << it->second.size() << " components"
83  << endl;
84  }
85 
86  pout() << " Summary: " << query.numComps() << " datasets at "
87  << query.m_num_points << " points" << endl;
88  }
89 
90  // Compute the bounds and spacings for each level
91  computeLevelLayouts();
92 
93  // Find boxes in which to interpolated each data point
94  InterpolationLayout interp_layout = findBoxes(query);
95 
96  // Prepare MPI buffers
97  prepareMPI(query, interp_layout);
98  // Calculate interpolated values
99  exchangeMPIQuery();
100  calculateAnswers(query);
101  exchangeMPIAnswer();
102 
103  int comp_idx = 0;
104 
105  for (typename InterpolationQuery::iterator deriv_it = query.compsBegin();
106  deriv_it != query.compsEnd(); ++deriv_it)
107  {
108  typedef std::vector<typename InterpolationQuery::out_t> comps_t;
109  comps_t &comps = deriv_it->second;
110 
111  for (typename comps_t::iterator it = comps.begin(); it != comps.end();
112  ++it)
113  {
114  double *out = it->second;
115  for (int point_idx = 0; point_idx < query.m_num_points; ++point_idx)
116  {
117  out[point_idx] =
118  m_query_data[comp_idx][m_mpi_mapping[point_idx]];
119  }
120  comp_idx++;
121  }
122  }

125 // Calculates each level's grid spacing and origin location
126 template <typename InterpAlgo>
127 void AMRInterpolator<InterpAlgo>::computeLevelLayouts()
128 {
129  ostream &_pout = pout();
130 
131  if (m_verbosity)
132  {
133  _pout << TAG << "Entering computeLevelLayouts" << endl;
134  }
135 
136  const Vector<AMRLevel *> &levels = const_cast<AMR &>(m_amr).getAMRLevels();
137  const int num_levels = m_num_levels; // levels.size();
138 
139  m_origin.resize(num_levels);
140  m_dx.resize(num_levels);
141 
142  IntVect prev_small_end;
143  IntVect prev_big_end;
144 
145  for (int level_idx = 0; level_idx < num_levels; ++level_idx)
146  {
147  AMRLevel &level = *levels[level_idx];
148  const Box &domain_box = level.problemDomain().domainBox();
149  const IntVect &small_end = domain_box.smallEnd();
150  const IntVect &big_end = domain_box.bigEnd();
151 
152  if (level_idx == 0)
153  {
154  m_origin[level_idx] = m_coarsest_origin;
155  m_dx[level_idx] = m_coarsest_dx;
156  }
157  else
158  {
159  const int ref_ratio = level.refRatio();
160 
161  for (int i = 0; i < CH_SPACEDIM; ++i)
162  {
163  // Sanity check
164  CH_assert(ref_ratio ==
165  (big_end[i] - small_end[i] + 1) /
166  (prev_big_end[i] - prev_small_end[i] + 1));
167 
168  // This level's dx
169  m_dx[level_idx][i] = m_dx[level_idx - 1][i] / ref_ratio;
170 
171  m_origin[level_idx][i] =
172  // coordinate of the previous level's origin
173  m_origin[level_idx - 1][i]
174  // ((edge of the boundary cell) + (distance away from
175  // boundary at the refinement level)) * (expressed in terms
176  // of previous level's grid coordinates)
177  + ((prev_small_end[i] - 0.5) + 1.0 / (2 * ref_ratio)) *
178  m_dx[level_idx - 1][i]
179  // offset from this level's grid origin to small end
180  - small_end[i] * m_dx[level_idx][i];
181  }
182  }
183 
184  if (m_verbosity >= 2)
185  {
186  _pout << " Level " << level_idx << "\t"
187  << "dx=(" << m_dx[level_idx][0] << "," << m_dx[level_idx][1]
188  << "," << m_dx[level_idx][2] << ")\t"
189  << "grid_origin=(" << m_origin[level_idx][0] << ","
190  << m_origin[level_idx][1] << "," << m_origin[level_idx][2]
191  << ")" << endl;
192  }
193 
194  prev_small_end = small_end;
195  prev_big_end = big_end;
196  }
197 
198  if (m_verbosity)
199  {
200  _pout << TAG << "Leaving computeLevelLayouts" << endl;
201  }

436 template <typename InterpAlgo>
437 void AMRInterpolator<InterpAlgo>::prepareMPI(InterpolationQuery &query,
438  const InterpolationLayout layout)
439 {
440  ostream &_pout = pout();
441 
442  if (m_verbosity)
443  {
444  _pout << TAG << "Entering prepareMPI" << endl;
445  }
446 
447  // Count the number of points queried to each rank
448  m_mpi.clearQueryCounts();
449  for (int point_idx = 0; point_idx < query.m_num_points; ++point_idx)
450  {
451  int rank = layout.rank[point_idx];
452  CH_assert(rank > -1);
453  m_mpi.incrementQueryCount(rank);
454  }
455 
456  // Resize MPI 'query' buffers
457  m_query_level.resize(query.m_num_points);
458  m_query_box.resize(query.m_num_points);
459  for (int i = 0; i < CH_SPACEDIM; ++i)
460  m_query_coords[i].resize(query.m_num_points);
461 
462  m_query_data.resize(query.numComps());
463  for (int comp = 0; comp < query.numComps(); ++comp)
464  m_query_data[comp].resize(query.m_num_points);
465 
466  // Reorder query data for MPI_Ialltoallv
467  m_mpi_mapping.resize(query.m_num_points);
468  std::vector<int> rank_counter(m_mpi.m_num_process, 0);
469  for (int point_idx = 0; point_idx < query.m_num_points; ++point_idx)
470  {
471  int rank = layout.rank[point_idx];
472  int idx = m_mpi.queryDispl(rank) + rank_counter[rank];
473 
474  m_query_level[idx] = layout.level_idx[point_idx];
475  m_query_box[idx] = layout.box_idx[point_idx];
476 
477  for (int i = 0; i < CH_SPACEDIM; ++i)
478  {
479  m_query_coords[i][idx] = query.m_coords[i][point_idx];
480  }
481 
482  m_mpi_mapping[point_idx] = idx;
483 
484  rank_counter[rank] += 1;
485  }
486 
487  m_mpi.exchangeLayout();
488 
489  // Resize MPI 'answer' buffers
490  int num_answers = m_mpi.totalAnswerCount();
491 
492  m_answer_level.resize(num_answers);
493  m_answer_box.resize(num_answers);
494  for (int i = 0; i < CH_SPACEDIM; ++i)
495  m_answer_coords[i].resize(num_answers);
496 
497  m_answer_data.resize(query.numComps());
498  for (int comp = 0; comp < query.numComps(); ++comp)
499  m_answer_data[comp].resize(num_answers);
500 
501  if (m_verbosity)
502  {
503  _pout << " Number of points that needs to be answered back:" << endl;
504  for (int rank = 0; rank < m_mpi.m_num_process; ++rank)
505  {
506  _pout << " Rank " << rank << "\t= " << m_mpi.queryCount(rank)
507  << endl;
508  }
509  _pout << TAG << "Leaving prepareMPI" << endl;
510  }

549 template <typename InterpAlgo>
550 void AMRInterpolator<InterpAlgo>::calculateAnswers(InterpolationQuery &query)
551 {
552  ostream &_pout = pout();
553 
554  if (m_verbosity)
555  {
556  _pout << TAG << "Entering calculateAnswer" << endl;
557  }
558 
559  const Vector<AMRLevel *> &levels = const_cast<AMR &>(m_amr).getAMRLevels();
560  // const int num_levels = levels.size();
561  // const int num_comps = query.numComps();
562  const int num_answers = m_mpi.totalAnswerCount();
563 
564  std::array<double, CH_SPACEDIM> grid_coord;
565  IntVect nearest;
566 
567  for (int answer_idx = 0; answer_idx < num_answers; ++answer_idx)
568  {
569  const int box_idx = m_answer_box[answer_idx];
570  const int level_idx = m_answer_level[answer_idx];
571 
572  const AMRLevel &level = *levels[level_idx];
573  const InterpSource &source = dynamic_cast<const InterpSource &>(level);
574  const LevelData<FArrayBox> &level_data = source.getLevelData();
575  const DisjointBoxLayout &box_layout = level_data.disjointBoxLayout();
576 
577  const Box &domain_box = level.problemDomain().domainBox();
578  const IntVect &small_end = domain_box.smallEnd();
579  const IntVect &big_end = domain_box.bigEnd();
580 
581  // Convert the LayoutIndex to DataIndex
582  const DataIndex data_idx(box_layout.layoutIterator()[box_idx]);
583 
584  const Box &box = box_layout[data_idx];
585  const FArrayBox &fab = level_data[data_idx];
586 
587  for (int i = 0; i < CH_SPACEDIM; ++i)
588  {
589  grid_coord[i] =
590  (m_answer_coords[i][answer_idx] - m_origin[level_idx][i]) /
591  m_dx[level_idx][i];
592 
593  if (!(grid_coord[i] >= box.smallEnd()[i] - 0.5 &&
594  grid_coord[i] <= box.bigEnd()[i] + 0.5))
595  {
596  std::ostringstream s;
597  s << "grid_coord[" << i << "] = " << grid_coord[i]
598  << " is out of range [" << (box.smallEnd()[i] - 0.5) << ","
599  << (box.bigEnd()[i] + 0.5) << "]";
600  MayDay::Abort(s.str().c_str());
601  }
602 
603  // point lies beyond the "small end" of the whole domain, but still
604  // within the boundary cell
605  if (grid_coord[i] < small_end[i] &&
606  grid_coord[i] >= small_end[i] - 0.5)
607  nearest[i] = small_end[i];
608 
609  // point lies beyond the "big end" of the whole domain, but still
610  // within the boundary cell
611  else if (grid_coord[i] > big_end[i] &&
612  grid_coord[i] <= big_end[i] + 0.5)
613  nearest[i] = big_end[i];
614 
615  // otherwise we round to nearest grid point
616  else
617  nearest[i] = (int)ceil(grid_coord[i] - 0.5);
618  }
619 
620  if (m_verbosity >= 2)
621  {
622  _pout << " Interpolating (";
623  for (int i = 0; i < CH_SPACEDIM; ++i)
624  {
625  _pout << m_answer_coords[i][answer_idx];
626  if (i < CH_SPACEDIM - 1)
627  _pout << ",";
628  }
629  _pout << ") in level " << level_idx << " box " << box_idx << endl;
