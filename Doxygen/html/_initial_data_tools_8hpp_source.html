15 // Convert a Tensor (with two lower indices) in spherical coords to cartesian
16 // coords
17 template <class data_t>
18 static Tensor<2, data_t>
19  spherical_to_cartesian_LL(Tensor<2, data_t> spherical_g, data_t x, double y,
20  double z)
21 {
22  Tensor<2, data_t> cartesian_g;
23 
24  // calculate useful position quantities
25  data_t rho2 = simd_max(x * x + y * y, 1e-12);
26  data_t rho = sqrt(rho2);
27  data_t r2 = simd_max(x * x + y * y + z * z, 1e-12);
28  data_t r = sqrt(r2);
29 
30  // And the sines and cosines of phi and theta
31  data_t cos_phi = x / rho;
32  data_t sin_phi = y / rho;
33 
34  // derivatives for jacobian matrix - drdx etc
35  Tensor<2, data_t> jac;
36  jac[0][0] = x / r;
37  jac[1][0] = cos_phi * z / r2;
38  jac[2][0] = -y / rho2;
39  jac[0][1] = y / r;
40  jac[1][1] = sin_phi * z / r2;
41  jac[2][1] = x / rho2;
42  jac[0][2] = z / r;
43  jac[1][2] = -rho / r2;
44  jac[2][2] = 0.0;
45 
46  // Convert the Tensor to cartesian coords
47  FOR2(i, j)
48  {
49  cartesian_g[i][j] = 0;
50  FOR2(k, m)
51  {
52  cartesian_g[i][j] += spherical_g[k][m] * jac[k][i] * jac[m][j];
53  }
54  }
55  return cartesian_g;
56 }

84  inv_jac[2][1] = -r * sin_theta;
85  inv_jac[0][2] = -y;
86  inv_jac[1][2] = x;
87  inv_jac[2][2] = 0.0;
88 
89  // transform the vector to cartesian coords

91  {
92  cartesian_v[i] = 0.0;
93  FOR1(j) { cartesian_v[i] += inv_jac[i][j] * spherical_v[j]; }
94  }
